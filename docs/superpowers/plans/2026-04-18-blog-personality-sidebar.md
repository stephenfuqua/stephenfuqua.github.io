# Blog Personality Sidebar Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the default Docusaurus blog sidebar with a personality block (avatar, bio, topic chips) and backfill a single bucket tag (`technology`, `nature`, or `bahai-faith`) onto every post in `blog/` so the chips link to non-empty tag pages.

**Architecture:** Swizzle `@docusaurus/theme-classic`'s `BlogSidebar` with a custom React component under `src/theme/BlogSidebar/`. Provide two one-shot Node scripts under `scripts/`: `classify-posts.js` produces a reviewable CSV of proposed tags; `apply-tags.js` reads the reviewed CSV and writes the chosen tag into each post's YAML frontmatter. Scripts are tested with Node's built-in `node:test` runner against a temporary `tmp-test-fixtures/` tree.

**Tech Stack:** Docusaurus 3.9 + React 19 (existing); Node.js `node:test` for script tests; `gray-matter` for frontmatter parsing; no additional CSS framework.

**Spec:** `docs/superpowers/specs/2026-04-18-blog-personality-sidebar-design.md`

Create `scripts/lib/csv.js`:

```js
// Minimal RFC-4180-ish CSV reader/writer. Assumes UTF-8 text input.
// Does not handle BOMs, unterminated quoted fields, or comments.

export function parseCsv(text) {
  const rows = [];
  let field = '';
  let row = [];
  let inQuotes = false;
  let i = 0;
  while (i < text.length) {
    const ch = text[i];
    if (inQuotes) {
      if (ch === '"') {
        if (text[i + 1] === '"') {
          field += '"';
          i += 2;
          continue;
        }
        inQuotes = false;
        i += 1;
        continue;
      }
      field += ch;
      i += 1;
      continue;
    }
    if (ch === '"') {
      inQuotes = true;
      i += 1;
      continue;
    }
    if (ch === ',') {
      row.push(field);
      field = '';
      i += 1;
      continue;
    }
    if (ch === '\n') {
      row.push(field);
      rows.push(row);
      field = '';
      row = [];
      i += 1;
      continue;
    }
    if (ch === '\r') {
      i += 1;
      continue;
    }
    field += ch;
    i += 1;
  }
  if (field.length > 0 || row.length > 0) {
    row.push(field);
    rows.push(row);
  }
  if (rows.length === 0) return [];
  const [header, ...data] = rows;
  return data.map((cols) => {
    const obj = {};
    header.forEach((key, idx) => {
      obj[key] = cols[idx] ?? '';
    });
    return obj;
  });
}

function quoteIfNeeded(value) {
  const s = value ?? '';
  if (/[",\n\r]/.test(s)) {
    return `"${s.replace(/"/g, '""')}"`;
  }
  return s;
}

export function stringifyCsv(rows, columns) {
  const header = columns.join(',');
  const body = rows
    .map((row) => columns.map((c) => quoteIfNeeded(row[c])).join(','))
    .join('\n');
  return `${header}\n${body}\n`;
}
```

```js
// Keyword-based post classifier. Deterministic, case-insensitive.
// Bucket precedence: nature > bahai-faith > technology (default).
// The author reviews the CSV output before any tags are applied,
// so 100% accuracy is not the goal — good-enough-to-reduce-tedium is.

const NATURE_PATTERNS = [
  /\bbird(s|ing|er|ers)?\b/i,
  /\bhik(e|es|ing)\b/i,
  /\btrail(s)?\b/i,
  /\bebird\b/i,
  /\bpark(s)?\b/i,
  /\bclimate\b/i,
  /\bwarbler(s)?\b/i,
  /\bflycatcher(s)?\b/i,
  /\bnature\b/i,
  /\bwildlife\b/i,
  /\bspecies\b/i,
  /\bforest(s)?\b/i,
  /\briver(s)?\b/i,
  /\bhornsby\b/i,
];

const FAITH_PATTERNS = [
  /bahá'í/i,
  /bahai/i,
  /abdu'l-bahá/i,
  /abdul-baha/i,
  /\bprayer\b/i,
  /\bspiritual\b/i,
  /\bwritings\b/i,
  /\bunity of religion\b/i,
  /\bfaith\b/i,
  /\bsoul\b/i,
  /\bgod\b/i,
];

function anyMatch(text, patterns) {
  return patterns.some((re) => re.test(text));
}

export function classify({ title, snippet }) {
  const text = `${title ?? ''}\n${snippet ?? ''}`;
  if (anyMatch(text, NATURE_PATTERNS)) return 'nature';
  if (anyMatch(text, FAITH_PATTERNS)) return 'bahai-faith';
  return 'technology';
}
```

```js
import matter from 'gray-matter';

// Adds `tag` to the `tags` array in the frontmatter of `raw` and returns the
// rewritten file text. Idempotent: if `tag` is already present, returns `raw`
// unchanged. If the file has no frontmatter, a minimal one is prepended.
export function addTag(raw, tag) {
  const parsed = matter(raw);
  const existing = Array.isArray(parsed.data.tags) ? parsed.data.tags : [];
  if (existing.includes(tag)) {
    return raw;
  }
  const newData = { ...parsed.data, tags: [...existing, tag] };
  return matter.stringify(parsed.content, newData);
}
```

```js
const BLOG_DIR = resolve('blog');
const OUTPUT = resolve('tag-classification.csv');
const SNIPPET_CHARS = 500;

async function walk(dir) {
  const out = [];
  for (const entry of await readdir(dir, { withFileTypes: true })) {
    const full = join(dir, entry.name);
    if (entry.isDirectory()) {
      out.push(...(await walk(full)));
    } else if (entry.isFile() && /\.(md|mdx)$/.test(entry.name)) {
      out.push(full);
    }
  }
  return out;
}

function snippet(body) {
  return body.replace(/\s+/g, ' ').trim().slice(0, SNIPPET_CHARS);
}

async function main() {
  const files = await walk(BLOG_DIR);
  const rows = [];
  for (const filepath of files) {
    const raw = await readFile(filepath, 'utf8');
    const parsed = matter(raw);
    const title = parsed.data.title ?? '';
    const snip = snippet(parsed.content);
    const proposed = classify({ title, snippet: snip });
    const currentTags = Array.isArray(parsed.data.tags)
      ? parsed.data.tags.join(';')
      : '';
    rows.push({
      filepath: relative(process.cwd(), filepath).replace(/\\/g, '/'),
      title,
      current_tags: currentTags,
      proposed_tag: proposed,
      first_paragraph_snippet: snip,
    });
  }
  rows.sort((a, b) => a.filepath.localeCompare(b.filepath));
  const csv = stringifyCsv(rows, [
    'filepath',
    'title',
    'current_tags',
    'proposed_tag',
    'first_paragraph_snippet',
  ]);
  await writeFile(OUTPUT, csv, 'utf8');
  console.log(`Wrote ${rows.length} rows to ${relative(process.cwd(), OUTPUT)}`);
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```

* [ ] **Step 1: Write the script**

Create `scripts/apply-tags.js`:

```js
#!/usr/bin/env node
// Reads tag-classification.csv and adds the proposed_tag to each post's
// frontmatter. Dry-run by default; pass --write to actually modify files.
import { readFile, writeFile } from 'node:fs/promises';
import { resolve } from 'node:path';
import { parseCsv } from './lib/csv.js';
import { addTag } from './lib/frontmatter.js';

const CSV_PATH = resolve('tag-classification.csv');
const ALLOWED = new Set(['technology', 'nature', 'bahai-faith']);
const write = process.argv.includes('--write');

function diffSummary(before, after) {
  if (before === after) return '(no change — tag already present)';
  const beforeLen = before.length;
  const afterLen = after.length;
  return `${beforeLen} -> ${afterLen} chars`;
}

async function main() {
  const csvRaw = await readFile(CSV_PATH, 'utf8');
  const rows = parseCsv(csvRaw);
  let changed = 0;
  let skipped = 0;
  let invalid = 0;
  for (const row of rows) {
    if (!ALLOWED.has(row.proposed_tag)) {
      console.error(`INVALID tag "${row.proposed_tag}" in ${row.filepath} — skipping`);
      invalid += 1;
      continue;
    }
    const path = resolve(row.filepath);
    const raw = await readFile(path, 'utf8');
    const next = addTag(raw, row.proposed_tag);
    if (next === raw) {
      skipped += 1;
      continue;
    }
    changed += 1;
    console.log(`${write ? 'WRITE' : 'DRY '} ${row.filepath} ${diffSummary(raw, next)}`);
    if (write) {
      await writeFile(path, next, 'utf8');
    }
  }
  console.log(
    `\n${write ? 'Wrote' : 'Would write'} ${changed} file(s); ${skipped} already tagged; ${invalid} invalid.`,
  );
  if (invalid > 0) process.exit(1);
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```
